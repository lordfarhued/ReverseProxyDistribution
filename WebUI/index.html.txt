        // –ü–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–µ —Ç–∞–±–æ–≤
        function switchTab(tabName) {
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
            
            event.target.classList.add('active');
            document.getElementById(tabName).classList.add('active');
            
            if (tabName === 'instances') loadInstances();
            if (tabName === 'blocking') loadBlockedIps();
        }
        
        // –ó–∞–≥—Ä—É–∑–∫–∞ –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–π
        async function loadConfigurations() {
            try {
                const response = await fetch(`${API_URL}/configuration`);
                const configs = await response.json();
                
                document.getElementById('totalConfigs').textContent = configs.length;
                
                const table = document.getElementById('configTable');
                if (configs.length === 0) {
                    table.innerHTML = '<tr><td colspan="7" style="text-align: center; padding: 30px;">–ù–µ—Ç –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–π</td></tr>';
                    return;
                }
                
                table.innerHTML = configs.map(c => `
                    <tr>
                        <td>${c.id}</td>
                        <td><strong>${c.key}</strong></td>
                        <td><span class="badge badge-${c.type}">${c.type}</span></td>
                        <td>${c.domain || '-'}</td>
                        <td>v${c.version}</td>
                        <td>${new Date(c.lastModified).toLocaleString('ru-RU')}</td>
                        <td>
                            <button class="btn" style="padding: 5px 15px; font-size: 12px;" onclick="deleteConfig(${c.id})">üóëÔ∏è</button>
                        </td>
                    </tr>
                `).join('');
            } catch (error) {
                console.error('Error loading configurations:', error);
                showAlert('configAlert', '–û—à–∏–±–∫–∞ –∑–∞–≥—Ä—É–∑–∫–∏ –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–π', 'error');
            }
        }
        
        // –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–∏
        async function addConfiguration() {
            const key = document.getElementById('configKey').value;
            const type = document.getElementById('configType').value;
            const domain = document.getElementById('configDomain').value;
            const value = document.getElementById('configValue').value;
            
            if (!key || !value) {
                showAlert('configAlert', '–ó–∞–ø–æ–ª–Ω–∏—Ç–µ –∫–ª—é—á –∏ –∑–Ω–∞—á–µ–Ω–∏–µ', 'error');
                return;
            }
            
            try {
                const response = await fetch(`${API_URL}/configuration`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ key, type, domain, value })
                });
                
                if (response.ok) {
                    showAlert('configAlert', '–ö–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏—è —É—Å–ø–µ—à–Ω–æ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∞!', 'success');
                    document.getElementById('configKey').value = '';
                    document.getElementById('configValue').value = '';
                    document.getElementById('configDomain').value = '';
                    loadConfigurations();
                }
            } catch (error) {
                showAlert('configAlert', '–û—à–∏–±–∫–∞ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è: ' + error.message, 'error');
            }
        }

        async function deleteConfig(id) {

                await fetch(`${API_URL}/configuration/${id}`, { method: 'DELETE' });

            } catch (error) {
                console.error('Error deleting config:', error);
            }
        }

        // –ë–ª–æ–∫–∏—Ä–æ–≤–∫–∞ IP
        async function blockIp() {
            const ipAddress = document.getElementById('blockIp').value;
            const reason = document.getElementById('blockReason').value;
            const expiresAt = document.getElementById('blockExpires').value;
            
            if (!ipAddress || !reason) {
                showAlert('blockAlert', '–ó–∞–ø–æ–ª–Ω–∏—Ç–µ IP –∏ –ø—Ä–∏—á–∏–Ω—É', 'error');

            try {
                const response = await fetch(`${API_URL}/configuration/block-ip`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        ipAddress,
                        reason,
                        expiresAt: expiresAt || null
                    })
                });
                
                if (response.ok) {
                    showAlert('blockAlert', `IP ${ipAddress} —É—Å–ø–µ—à–Ω–æ –∑–∞–±–ª–æ–∫–∏—Ä–æ–≤–∞–Ω!`, 'success');
                    document.getElementById('blockIp').value = '';
                    document.getElementById('blockReason').value = '';
                    document.getElementById('blockExpires').value = '';
                    loadBlockedIps();
                }
            } catch (error) {
                showAlert('blockAlert', '–û—à–∏–±–∫–∞ –±–ª–æ–∫–∏—Ä–æ–≤–∫–∏: ' + error.message, 'error');
            }
        }
